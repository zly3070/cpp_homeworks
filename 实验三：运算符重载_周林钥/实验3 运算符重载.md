### 1. 复数类运算符重载（test1.cpp）

#### （1）程序代码

```cpp
#include <bits/stdc++.h>
using namespace std;

class Complex{
		private:
			double real;
			double imag;

		public:
			Complex() {real = 0;imag = 0;}
			Complex(double r, double i) : real(r), imag(i) {}
			void getComplex();
			Complex operator+(Complex c2) const;
			Complex operator-(Complex c2) const;
			Complex operator*(Complex c2) const;
			Complex operator/(Complex c2) const;
	};
	
void Complex::getComplex(){
	if (real == 0){
		if (imag != 0) cout << imag << "i" << endl;		 
		else cout << "0" << endl;		
	} else {
		if (imag != 0){
			if (imag < 0) cout << real << imag << "i" << endl;
			else cout << real << "+" << imag << "i" << endl;
		} else cout << real << endl;
	}
}

Complex Complex::operator+(const Complex c) const {
	Complex tmp;
	tmp.real = real + c.real;
	tmp.imag = imag + c.imag;
	return tmp;
}

Complex Complex::operator-(const Complex c) const {
	Complex tmp;
	tmp.real = real - c.real;
	tmp.imag = imag - c.imag;
	return tmp;
}

Complex Complex::operator*(const Complex c) const {
	Complex tmp;
	tmp.real = real*c.real - imag*c.imag;
	tmp.imag = real*c.imag + imag*c.real;
	return tmp;
}

Complex Complex::operator/(const Complex c) const {
	double denominator = c.real * c.real + c.imag * c.imag;
    double realPart = (real * c.real + imag * c.imag) / denominator;
    double imagPart = (imag * c.real - real * c.imag) / denominator;
    
    return Complex(realPart, imagPart);
}
	
int main(){
	Complex c1(-1, 1);
	Complex c2(3, 2);
	Complex c3 = c1 + c2;
	Complex c4 = c1 - c2;
	Complex c5 = c1 * c2;
	Complex c6 = c1 / c2;
	c6.getComplex();
	return 0;
}
```

#### （2）运行结果

```
-0.0769231-0.384615i
```

#### （3）结果分析

通过重载 +、-、*、/ 运算符，Complex 类能够像内置类型一样进行四则运算。运算符重载函数被定义为 const 成员函数，确保不修改当前对象。除法运算通过分母有理化实现，结果以浮点数形式输出。

## 2. 复数类的混合运算（test2.cpp

### （1）程序代码

```cpp
#include <bits/stdc++.h>
using namespace std;

class Complex {
private:
    double real;
    double imag;

public:
    Complex() { real = 0; imag = 0; }
    Complex(double r, double i) : real(r), imag(i) {}
    Complex(double r) : real(r), imag(0) {}  // 转换构造函数
    void getComplex();
    friend Complex operator+(const Complex& jiashu1, const Complex& jiashu2);
};

void Complex::getComplex() {
    if (real == 0) {
        if (imag != 0) cout << imag << "i" << endl;
        else cout << "0" << endl;
    } else {
        if (imag != 0) {
            if (imag < 0) cout << real << imag << "i" << endl;
            else cout << real << "+" << imag << "i" << endl;
        } else cout << real << endl;
    }
}

Complex operator+(const Complex& jiashu1, const Complex& jiashu2) {
    return Complex(jiashu1.real + jiashu2.real, jiashu1.imag + jiashu2.imag);
}

int main() {
    Complex c1(-1, 1);
    Complex c2(3, 2);
    int int1 = 4;
    
    Complex jiahe1 = int1 + c2;   // int + Complex
    Complex jiahe2 = c2 + int1;   // Complex + int
    Complex jiahe3 = c1 + c2;     // Complex + Complex
    
    jiahe1.getComplex();
    jiahe2.getComplex();
    jiahe3.getComplex();
    
    return 0;
}
```

### （2）运行结果

```text
7+2i
7+2i
2+3i
```

### （3）结果分析

- 转换构造函数 `Complex(double r)` 使得 `int` 类型可以隐式转换为 `Complex` 类型。
- 友元函数 `operator+` 提供了 `Complex + Complex` 的加法规则。
- 当遇到 `int + Complex` 或 `Complex + int` 时，编译器会优先尝试将 `int` 转换为 `Complex`，然后调用友元 `operator+`。
- 这种设计使得 `Complex` 类能够与内置数值类型进行混合运算，提高了类的易用性。

### 3. 矩阵类的设计与运算符重载（test3.cpp）

#### （1）程序代码

```cpp
#include <bits/stdc++.h>
using namespace std;

class Matrix{
	private:
		array<array<int, 3>, 2> data = {};
	public:
		Matrix(){
			for (int i = 0; i < 2; i ++){
				for (int j = 0; j < 3; j++){
					data[i][j] = 0;
				}
			}
		}
		Matrix(initializer_list<initializer_list<int>> list){
			int i = 0;
			for (const auto& row : list){
				int j = 0;
				for (const auto& val : row){
					data[i][j] = val;
					j ++;
				}
				i ++;
			}
		}
		friend Matrix operator+(const Matrix& juzhen1, const Matrix& juzhen2);
		void getMatrix() const;
		friend istream& operator>>(istream&, Matrix&);
		friend ostream& operator<<(ostream&, const Matrix&);
};

Matrix operator+(const Matrix& juzhen1, const Matrix& juzhen2){
	Matrix tmp;
	for (int i = 0; i < 2; i++){
		for (int j = 0; j < 3; j++){
			tmp.data[i][j] = juzhen1.data[i][j] + juzhen2.data[i][j];
		}
	}
	return tmp;
}

void Matrix::getMatrix() const {
	for (int i = 0; i < 2; i++){
		for (int j = 0; j < 3; j++){
			cout << data[i][j] << " ";
		}
		cout << endl;
	}
}

istream& operator>>(istream& input, Matrix& m){
    cout << "请输入6个整数（2行3列）：" << endl;
    for (int i = 0; i < 2; i++){
        for (int j = 0; j < 3; j++){
            input >> m.data[i][j];
        }
    }
    return input;
}

ostream& operator<<(ostream& output, const Matrix& m){
    for (int i = 0; i < 2; i++){
        for (int j = 0; j < 3; j++){
            output << m.data[i][j] << " ";
        }
        output << endl;
    }
    return output;
}

int main(){
	Matrix a = {
		{2, 3, 0},
		{1, 4, 5}
	}; 
	Matrix b = {
		{1, 2, 3},
		{-1, 2, 4}
	}; 
	Matrix c = a + b;
	cout << c << endl;
	Matrix d;
	cin >> d; 
	cout << d;
	return 0;
}
```

#### （2）运行结果

```
3 5 3
0 6 9
```

#### （3）结果分析

- 使用 `initializer_list` 实现了矩阵的列表初始化。
- 重载了 `+` 运算符，实现矩阵加法。
- 重载了 `>>` 和 `<<` 运算符，支持直接使用 `cin` 和 `cout` 进行矩阵的输入输出。
- 友元函数的声明使得运算符重载函数可以访问私有成员 `data`。

### 5. 学生与教师类的友元设计（test5.cpp）

#### （1）程序代码

```cpp
#include <bits/stdc++.h>
using namespace std;

class Student {
private:
    int num;
    string name;
    bool sex;
public:
    Student(int n, string na, bool s) : num(n), name(na), sex(s) {}
	friend class Teacher;
    friend void getInfo(const Student&);
};

class Teacher {
private:
    int num;
    string name;
    bool sex;
public:
    Teacher(const Student& stu) : num(stu.num), name(stu.name), sex(stu.sex) {}
    
    friend void getInfo(const Teacher&);
};

void getInfo(const Teacher& obj) {
    cout << "学号/工号：" << obj.num << endl;
    cout << "姓名：" << obj.name << endl;
    cout << "性别：" << (obj.sex ? "男" : "女") << endl;
}

void getInfo(const Student& obj) {
    cout << "学号/工号：" << obj.num << endl;
    cout << "姓名：" << obj.name << endl;
    cout << "性别：" << (obj.sex ? "男" : "女") << endl;
}

int main() {
    Student stu(123, "张三", true);
    Teacher tea(stu);
    
    getInfo(stu);
    getInfo(tea);
    
    return 0;
}
```

#### （2）运行结果

```
学号/工号：123
姓名：张三
性别：男

学号/工号：123
姓名：张三
性别：男
```

#### （3）结果分析

- `friend class Teacher` 使得 Teacher 类可以访问 Student 的私有成员，从而在构造函数中直接使用 `stu.num`、`stu.name` 等。
- `friend void getInfo(...)` 使得全局函数可以访问 Student 和 Teacher 的私有成员，实现统一的信息输出。
- 友元机制在需要跨类访问私有数据时非常有用，但会破坏封装性，应谨慎使用。
